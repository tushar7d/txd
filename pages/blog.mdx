import { getPagesUnderRoute } from "nextra/context";
import Link from "next/link";

export function BlogHeader() {
  return (
    <div className="max-w-screen-lg mx-auto pt-4 pb-8 mb-16  mt-24">
      
       <div className="font-bold leading-tight font-serif text-5xl text-center ">Writing</div>
      
    
    </div>
  );
}

export function BlogIndex() {
  return getPagesUnderRoute("/blog").map((page) => {
    return (
      <div key={page.route} className="mb-10 p-4  border-b  border-white">
        <Link href={page.route} >

        <h1 className="text-2xl">
          {page.meta?.title || page.frontMatter?.title || page.name}
       </h1>
        <p className="opacity-80">
          {page.frontMatter?.description}
        </p>
        {page.frontMatter?.date ? (
          <p className="opacity-50 text-sm">{page.frontMatter.date}</p>
        ) : null}
         </Link>
      </div>
    );
  });
}

<BlogHeader />
<BlogIndex />
